def execute
  raise MissingBitmap if app.bitmap.nil?

  width = @app.bitmap.width
  height = @app.bitmap.height
  (1..height).each do |y|
    (1..width).each do |x|
      print @app.bitmap[x, y]
    end
    puts
  end
end
